\begin{frame}[c]{The traveling salesperson problem (TSP)}
  \begin{itemize}
  \item \structure{Problem Instance} A set of cities and distances among them,

    or simply a weighted graph
    \smallskip
  \item \structure{Problem Class} What is the shortest possible route visiting each city once and returning to the city of origin?
    \bigskip
  \item<2-> \structure{Note}
    \begin{itemize}
    \item TSP extends the Hamiltonian cycle problem:
      \par
      Is there a cycle in a graph visiting each node exactly once
      \smallskip
    \item TSP is relevant to applications in logistics, planning, chip design,
      and the core of the vehicle routing problem
    \end{itemize}
  \end{itemize}
\end{frame}
