\begin{frame}[fragile]{Satisfiability testing}

  \begin{itemize}
  \item<1-> \structure{Problem Instance} \ A propositional formula $\phi$ in CNF
  \item<1-> \structure{Problem Class}    \ Is there an assignment of propositional variables to true
    and false such that a given formula $\phi$ is true

    \smallskip

  \item<2-> \structure{Example} \ Consider formula
    \[
      \alt<6>{(\neg a\wedge b\to\bot)\wedge (a\wedge\neg b\to\bot)}%
             {\alt<5>{\neg(\neg a\wedge b)\wedge\neg(a\wedge\neg b)}%
                     {(a\vee\neg b)\wedge(\neg a\vee b)}}
    \]

  \item<3-> \structure{Logic Program}
    \smallskip

    \begin{tabular}{lp{0em}lp{1em}l}
      \text{\textbf{Generator}}
      &&
      \text{\textbf{Tester}}
      &&
      \text{\textbf{Stable models}}
      \\[3pt]
      \begin{tabular}[t]{l}
        \texttt{\{a\}.}\\
        \texttt{\{b\}.}
      \end{tabular}
      &&
      \begin{tabular}[t]{l}
        \texttt{:- not a,     b.}\\
        \texttt{:-     a, not b.}
      \end{tabular}
      &&
      \begin{tabular}[t]{rcl}
        $X_1$&=&$\{\mathtt{a},\mathtt{b}\}$\\
        $X_2$&=&$\{                     \}$
      \end{tabular}
    \end{tabular}
    \medskip
  \item<4->  \structure{Note} \ The generator puts \texttt{a} and \texttt{b} under the open world assumption
  \item<4->[]  \phantom{Note} \ The tester eliminates interpretations; it is expressed negatively
  \end{itemize}
\end{frame}
