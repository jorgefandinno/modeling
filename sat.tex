\begin{frame}[c]{Satisfiability testing}

  \begin{itemize}
  \item \structure{Problem Instance} A propositional formula $\phi$ in CNF
  \item \structure{Problem Class} Is there an assignment of propositional variables to true
    and false such that a given formula $\phi$ is true

    \pause
    \bigskip

  \item \structure{Example:} Consider formula
    \[
    \alert<4>{(a\vee\neg b)}\wedge\alert<5>{(\neg a\vee b)}
    \]

  \item \structure{Logic Program} \ \bigskip
    \begin{center}
    \begin{tabular}{lp{0em}lp{1em}l}
      \text{\textbf{Generator}}
      &&
      \text{\textbf{Tester}}
      &&
      \text{\textbf{Stable models}}
      \\
      \begin{tabular}[t]{lcl}
        \alert<3>{$\{\,a\,\}$}    &\alert<3>{$\leftarrow$}&\\
        \alert<3>{$\{\,b\,\}$}    &\alert<3>{$\leftarrow$}&
      \end{tabular}
      &&
      \begin{tabular}[t]{cl}
        \alert<4>{$\leftarrow$}&\alert<4>{$\neg a,     b $}\\
        \alert<5>{$\leftarrow$}&\alert<5>{$     a, \neg b$}
      \end{tabular}
      &&
      \begin{tabular}[t]{rcl}
        X$_1$&=&$\{a,b\}$\\
        X$_2$&=&$\{   \}$
      \end{tabular}
    \end{tabular}
    \end{center}
  \end{itemize}
\end{frame}
